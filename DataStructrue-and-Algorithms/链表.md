
#### 链表反转
1. 头插法
```Go
/**
 *	示意 ：
 *	输入: 1->2->3->NULL
 *	过程 (添加虚拟头节点 -1):
 * 	   -1->1->NULL	 2->3->null
 *	   -1->2->1->NULL   3->null 
 *	   -1->3->2->1->NULL
 *	输出: 3->2->1->NULL 
 *
 * Definition for singly-linked list.
 * type ListNode struct {
 *     Val int
 *     Next *ListNode
 * }
 */
func reverseList(head *ListNode) *ListNode {
    newHead  := ListNode{-1,nil}
    for head != nil{
        next := head.Next
        head.Next = newHead.Next
        newHead.Next = head
        head = next
    }
    return newHead.Next
}
```
