
##### 链表反转
1. 头插法

```Go
/**
 * Definition for singly-linked list.
 * type ListNode struct {
 *     Val int
 *     Next *ListNode
 * }
 */
func reverseList(head *ListNode) *ListNode {
    newHead  := ListNode{-1,nil}
    for head != nil{
        next := head.Next
        head.Next = newHead.Next
        newHead.Next = head
        head = next
    }
    return newHead.Next
}
```
