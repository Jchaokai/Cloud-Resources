
##### 链表反转
1. 头插法
&emsp;示意：
&emsp;&emsp;输入: 1->2->3->4->5->NULL
&emsp;&emsp;guoch
&emsp;&emsp;输出: 5->4->3->2->1->NULL
```Go
/**
 * Definition for singly-linked list.
 * type ListNode struct {
 *     Val int
 *     Next *ListNode
 * }
 */
func reverseList(head *ListNode) *ListNode {
    newHead  := ListNode{-1,nil}
    for head != nil{
        next := head.Next
        head.Next = newHead.Next
        newHead.Next = head
        head = next
    }
    return newHead.Next
}
```
